313  */

316     var temp = properDate[2];
317     properDate[2] = properDate[1];
318     properDate[1] = temp;

359         beforeShowDay: function (date) {

361             console.log(date);
362             for (var i = 0; i < myDateArray.length; i++) {
363                 if (date === myDateArray[i].eventDate) {
364                     found = true;
365                 }
366             }
367             if (found) {
368                 found = false;
369                 return [true, 'eventDate', 'eventExists'];
370             } else {
371                 return [true, '', ''];
372             }
373         },
374         onSelect: function () {
375             $(this).change();
376         }
377     });
378     $('#calendarDate').on('change', function () {

598                 }
599                 if (startupLoginCheck === "nick" && firstLoad) {

609                 }
610                 firstLoad = false;
611             } else {
612                 stillLogged = false;
613                 firstLoad = false;
614             }
615         }

620  * Handles user logins to make sure only one person is logged in at a

622  */

624     var d = new Date,
625         dateFormatted = [d.getFullYear(),
626         d.getMonth() + 1,
627         d.getDate()].join('-');
628 
629     pageLoad();
630 
631     if (!stillLogged || loggedIn === selectedId) {

633             if (loggedIn === "nick") {
634                 user1Id.style.backgroundColor = "white";
635                 registerUserLogin(selectedId, "1", dateFormatted);
636                 successfullyLoggedOut();
637             } else {
638                 registerUserLogin(selectedId, "0", dateFormatted);
639                 user1Id.style.backgroundColor = "#59dc51";
640                 successfullyLoggedIn();
641                 loggedIn = "nick";
642             }
643         } else if (selectedId === "libbie") {
644             if (loggedIn === "libbie") {
645                 user2Id.style.backgroundColor = "white";
646                 registerUserLogin(selectedId, "1", dateFormatted);
647                 successfullyLoggedOut();
648             } else {
649                 registerUserLogin(selectedId, "0", dateFormatted);
650                 user2Id.style.backgroundColor = "#59dc51";
651                 successfullyLoggedIn();
652                 loggedIn = "libbie";
653             }
654         }
655     } else {
656         document.getElementById("loginText").appendChild(document.createTextNode("Sorry, " + toTitleCase(loggedIn) + " is still logged in"));
657         setTimeout(function () {
658             document.getElementById("loginText").removeChild(document.getElementById("loginText").firstChild);

696 function isValidDate(dateString) {
697     var regEx = /^\d{4}-\d{2}-\d{2}$/;
698     if (!dateString.match(regEx)) return false;  // Invalid format
699     var d = new Date(dateString);
700     if (!d.getTime() && d.getTime() !== 0) return false; // Invalid date
701     return d.toISOString().slice(0, 10) === dateString;
702 }
